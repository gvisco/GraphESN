 12      """Perform the iterative Least Mean Squares algorithm. 
 13   
 14      The algorithm performs a gradient descent to minimize the sum of the  
 15      squared errors.     
 16   
 17      Arguments: 
 18      network     --  Model to be trained. 
 19      dataset     --  Training-set (list of g_esn.graph.Graph) 
 20      label       --  Name of the input label (string). 
 21      Target      --  Target matrix (scipy.array (len(dataset), Ny)) 
 22      eta         --  Learning rate (real) 
 23      momentum    --  Momentum (real) 
 24      threshold   --  Minimum variation to consider the algorithm not converged  
 25                      yet (real). 
 26      patience    --  Patience factor (int) 
 27      maxit       --  Maximum iterations allowed (int) 
 28      decay       --  Weight decay factor. 
 29   
 30      Return: 
 31      sq_error    --  Last squared error (real). 
 32      epochs      --  Number of iteration performed (int). 
 33      W           --  Obtained weights matrix (scipy.array (Ny, Nf)) 
 34   
 35      """ 
 36      # collect the input 
 37      Input = scipy.array([network.get_activations(g, label) for g in dataset]) 
 38      Input = scipy.concatenate((Input, scipy.ones((Input.shape[0],1))), 1) # bias 
 39      # some check 
 40      Ns = len(dataset) # number of samples 
 41      Ny = Target.shape[1] # number of outputs 
 42      Nf = Input.shape[1] # number of features (including the bias) 
 43      assert Input.shape == (Ns, Nf), "Invalid shape. I:%s" % str(Input.shape) 
 44      assert Target.shape == (Ns, Ny), "Invalid shape. E:%s" % str(Target.shape) 
 45      assert (not scipy.isnan(Input).any()) and (not scipy.isinf(Input).any()),\ 
 46                  "Some input is NaN or Infty" 
 47      assert (not scipy.isnan(Target).any()) and (not scipy.isinf(Target).any()),\ 
 48                  "Some target is NaN or Infty" 
 49      # perform iterative fitting 
 50      best_error = float_info.max 
 51      quit_epoch = patience 
 52      epoch = 0 
 53      last_delta = scipy.zeros((1,Nf)) # last computed weight change 
 54      W = scipy.random.uniform(-1.0, 1.0, network.W_out.shape) # (Ny,Nf) 
 55      Net = scipy.dot(Input, W.T) # (Ns,Ny) 
 56      Out = network.out_act(Net) # (Ns,Ny) 
 57      while epoch < quit_epoch and epoch < maxit: 
 58          # slope: \partial E / \partial w         
 59          primeF = network.out_act.der(Net) # (Ns,Ny) 
 60          assert (not scipy.isnan(primeF).any()) and (not scipy.isinf(primeF).any()),\ 
 61                      "Some derivative is NaN or Infty" 
 62          EP = (Target - Out) * primeF # (Ns,Ny) 
 63          slope = []  
 64          for o in xrange(Ny): 
 65              sl_o = [] 
 66              for j in xrange(Nf): 
 67                  sl_o.append(-scipy.sum(EP[:,o] * Input[:,j])) 
 68              slope.append(sl_o) 
 69          # weights update         
 70          gradient = - scipy.array(slope) # (Ny,Nf) 
 71          delta = (gradient * eta) + (last_delta * momentum) 
 72          W = (W + delta) - (W * decay) 
 73          last_delta = delta 
 74          # current error 
 75          Net = scipy.dot(Input, W.T) # (Ns,Ny) 
 76          Out = network.out_act(Net) # (Ns,Ny) 
 77          errors = Target - Out 
 78          sq_error = scipy.sum(0.5 * (errors**2)) 
 79          # update stop condition 
 80          if (best_error - sq_error) / sq_error > threshold: 
 81              best_error = sq_error 
 82              quit_epoch = epoch + patience 
 83          epoch += 1 
 84      # replace current readout 
 85      network.W_out = W 
 86      return sq_error, epoch, W
